get.aplha <- function(mat.x, vec.y){
    ones <- rep(c(1),nrow(mat.x))
    mat.x <- cbind(ones,mat.x)
    x.T.x <- solve(t(mat.x) %*% mat.x)
    all.x <- x.T.x %*% t(mat.x)
    alpha <- all.x %*% vec.y
    return(alpha)
}

loss.1 <- function(vec.y, mat.x, a){
    ones <- rep(c(1),nrow(mat.x))
    mat.x <- cbind(ones,mat.x)
    y.x.a <- (vec.y - mat.x %*% a)
    my.loss <- t(y.x.a) %*% y.x.a
    return(my.loss)
}
